  1. start hadoop

    sh bin/start-all.sh

  2. exec wordcount

    1. put input into hdfs

      ./bin/hadoop dfs -put input/a input

    2. exec

      ./bin/hadoop jar example.jar wordcount input output

    3. get output from hdfs

      ./bin/hadoop dfs -get output/part-r-00000 .
