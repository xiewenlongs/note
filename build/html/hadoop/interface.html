FileStatus[] inputFiles = local.listStatus (inputDir); FSDataOutputStream out = hdfs.create (hdfsFile);

for (int i = 0; i < inputFiles.length; i++) {

FSDataInputStream in = local.open (inputFiles[i].getPath ()); byte buffer[] = new byte [256]; int bytesRead = 0; while ((bytesRead = in.read (buffer)) > 0) {

} in.close (); System.out.println (“aaaa:” + inputFiles[i].getPath ());

} out.close (); System.out.println(new Person().talk()) ;

}

}
